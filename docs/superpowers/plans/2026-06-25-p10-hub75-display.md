# P10 HUB75 Display Support — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add HUB75 P10 (128×32) driver behind an `IDisplayDriver` interface while keeping the MAX7219 Wokwi simulation fully working.

**Architecture:** `DisplayManager` is decoupled from hardware via `IDisplayDriver`. `Hub75Driver` wraps `ESP32-HUB75-MatrixPanel-I2S-DMA` for the real panel; `Max7219Driver` wraps MD_Parola for the Wokwi sim. `main.cpp` picks the driver at compile time via `-D USE_HUB75` / `-D USE_MAX7219`.

**Tech Stack:** PlatformIO, ESP32 Arduino framework, MD_Parola/MD_MAX72XX, ESP32-HUB75-MatrixPanel-I2S-DMA, Adafruit GFX, Wokwi simulator.

// src/Max7219Driver.cpp
#ifdef USE_MAX7219
#include "Max7219Driver.h"

Max7219Driver::Max7219Driver(uint8_t csPin, uint8_t numDevices)
  : _display(MD_MAX72XX::PAROLA_HW, csPin, numDevices)
{}

void Max7219Driver::begin() {
  _display.begin();
  _display.setIntensity(4);
  _display.displayClear();
}

void Max7219Driver::clear() {
  _display.displayClear();
  _currentText = "";
}

void Max7219Driver::showRow(uint8_t row, const char* text) {
  if (row != 1) return;
  String s(text);
  if (s == _currentText) return;
  _currentText = s;
  s.toCharArray(_textBuf, sizeof(_textBuf));
  if (s.length() <= 5) {
    _display.displayZoneText(0, _textBuf, PA_CENTER, 60, 500, PA_PRINT, PA_NO_EFFECT);
  } else {
    _display.displayZoneText(0, _textBuf, PA_LEFT, 60, 0, PA_SCROLL_LEFT, PA_SCROLL_LEFT);
  }
  _display.synchZoneStart();
}

void Max7219Driver::update() {
  _display.displayAnimate();
}

#endif // USE_MAX7219
```

### Task 3: Hub75Driver

// src/Hub75Driver.h
#pragma once
#ifdef USE_HUB75

#include "IDisplayDriver.h"
#include <ESP32-HUB75-MatrixPanel-I2S-DMA.h>

#define HUB75_PANEL_WIDTH   128
#define HUB75_PANEL_HEIGHT  32
#define HUB75_CHAIN         1

#define HUB75_R1_PIN   25
#define HUB75_G1_PIN   26
#define HUB75_B1_PIN   27
#define HUB75_R2_PIN   14
#define HUB75_G2_PIN   12
#define HUB75_B2_PIN   13
#define HUB75_A_PIN    23
#define HUB75_B_PIN    22
#define HUB75_C_PIN    5
#define HUB75_D_PIN    17
#define HUB75_CLK_PIN  16
#define HUB75_LAT_PIN  4
#define HUB75_OE_PIN   15

class Hub75Driver : public IDisplayDriver {
public:
  Hub75Driver();
  ~Hub75Driver() override;
  void begin() override;
  void clear() override;
  void showRow(uint8_t row, const char* text) override;
  void update() override;

private:
  MatrixPanel_I2S_DMA* _matrix;

  static const uint8_t kNumRows = 3;
  static const uint8_t kRowY[kNumRows];
  static const uint8_t kRowH = 10;

  void drawCenteredText(const char* text, uint8_t y, uint16_t color);
};

#endif // USE_HUB75
```

- [ ] **Step 2: Write Hub75Driver.cpp**

```cpp
// src/Hub75Driver.cpp
#ifdef USE_HUB75
#include "Hub75Driver.h"

const uint8_t Hub75Driver::kRowY[Hub75Driver::kNumRows] = {1, 12, 23};

Hub75Driver::Hub75Driver() : _matrix(nullptr) {}

Hub75Driver::~Hub75Driver() { delete _matrix; }

void Hub75Driver::begin() {
  HUB75_I2S_CFG::i2s_pins pins = {
    HUB75_R1_PIN, HUB75_G1_PIN, HUB75_B1_PIN,
    HUB75_R2_PIN, HUB75_G2_PIN, HUB75_B2_PIN,
    HUB75_A_PIN, HUB75_B_PIN, HUB75_C_PIN, HUB75_D_PIN, -1,
    HUB75_LAT_PIN, HUB75_OE_PIN, HUB75_CLK_PIN
  };
  HUB75_I2S_CFG mxconfig(HUB75_PANEL_WIDTH, HUB75_PANEL_HEIGHT, HUB75_CHAIN, pins);
  _matrix = new MatrixPanel_I2S_DMA(mxconfig);
  if (!_matrix->begin()) {
    Serial.println("HUB75: matrix init failed");
  }
  _matrix->setBrightness8(200);
  _matrix->clearScreen();
  _matrix->setTextSize(1);
  _matrix->setTextWrap(false);
}

void Hub75Driver::clear() {
  if (_matrix) _matrix->clearScreen();
}

void Hub75Driver::showRow(uint8_t row, const char* text) {
  if (row >= kNumRows || !_matrix) return;
  uint8_t y = kRowY[row];
  _matrix->fillRect(0, y, HUB75_PANEL_WIDTH, kRowH, 0);
  if (text && text[0] != '\0') {
    drawCenteredText(text, y, _matrix->color565(255, 255, 255));
  }
}

void Hub75Driver::update() {}

void Hub75Driver::drawCenteredText(const char* text, uint8_t y, uint16_t color) {
  int16_t x1, y1;
  uint16_t w, h;
  _matrix->getTextBounds(text, 0, y, &x1, &y1, &w, &h);
  int16_t x = ((int16_t)HUB75_PANEL_WIDTH - (int16_t)w) / 2;
  if (x < 0) x = 0;
  _matrix->setTextColor(color);
  _matrix->setCursor(x, y);
  _matrix->print(text);
}

#endif // USE_HUB75
```

```cpp
// src/DisplayManager.cpp
#include "DisplayManager.h"

DisplayManager::DisplayManager(IDisplayDriver& driver, IDataProvider& provider)
  : _driver(driver), _provider(provider), _lastRender(0)
{}

void DisplayManager::begin() {
  _driver.begin();
  _driver.showRow(0, "");
  _driver.showRow(1, "INIT");
  _driver.showRow(2, "");
}

void DisplayManager::update() {
  _driver.update();
  DisplayState state = _provider.getCurrentState();
  unsigned long now = millis();
  bool statusChanged = (state.status != _lastStatus);
  bool needsRefresh = isTimeBased(state.status) && (now - _lastRender >= 1000);
  if (statusChanged || needsRefresh) {
    _lastStatus = state.status;
    _lastRender = now;
    render(state);
  }
}

void DisplayManager::render(const DisplayState& state) {
  if      (state.status == "IDLE")        renderIdle(state);
  else if (state.status == "QUEUE")       renderQueue(state);
  else if (state.status == "PREPARING")   renderPreparing(state);
  else if (state.status == "RUNNING")     renderRunning(state);
  else if (state.status == "COOLDOWN")    renderCooldown(state);
  else if (state.status == "MAINTENANCE") renderMaintenance(state);
  else if (state.status == "OFFLINE")     renderOffline(state);
}

bool DisplayManager::isTimeBased(const String& status) const {
  return status == "RUNNING" || status == "COOLDOWN" || status == "PREPARING";
}

void DisplayManager::renderIdle(const DisplayState& state) {
  _driver.showRow(0, "WELCOME");
  _driver.showRow(1, "PICKLEBALL");
  _driver.showRow(2, "COURT");
}

void DisplayManager::renderQueue(const DisplayState& state) {
  String row0 = "C"; row0 += state.courtNumber; row0 += " "; row0 += state.matchType;
  String row1 = "";
  for (uint8_t i = 0; i < state.team1Count; i++) { if (i > 0) row1 += "/"; row1 += state.team1[i]; }
  String row2 = "VS ";
  for (uint8_t i = 0; i < state.team2Count; i++) { if (i > 0) row2 += "/"; row2 += state.team2[i]; }
  _driver.showRow(0, row0.c_str());
  _driver.showRow(1, row1.c_str());
  _driver.showRow(2, row2.c_str());
}

void DisplayManager::renderPreparing(const DisplayState& state) {
  String row0 = "COURT "; row0 += state.courtNumber;
  _driver.showRow(0, row0.c_str());
  _driver.showRow(1, state.remainingTime.c_str());
  _driver.showRow(2, "PREPARING");
}

void DisplayManager::renderRunning(const DisplayState& state) {
  String row0 = "C"; row0 += state.courtNumber; row0 += " "; row0 += state.matchType;
  _driver.showRow(0, row0.c_str());
  _driver.showRow(1, state.remainingTime.c_str());
  _driver.showRow(2, "RUNNING");
}

void DisplayManager::renderCooldown(const DisplayState& state) {
  _driver.showRow(0, "GAME DONE");
  _driver.showRow(1, state.remainingTime.c_str());
  _driver.showRow(2, "NEXT MATCH");
}

void DisplayManager::renderMaintenance(const DisplayState& state) {
  _driver.showRow(0, "COURT CLOSED");
  _driver.showRow(1, "MAINTENANCE");
  _driver.showRow(2, "");
}

void DisplayManager::renderOffline(const DisplayState& state) {
  _driver.showRow(0, "SERVER");
  _driver.showRow(1, "OFFLINE");
  _driver.showRow(2, "RECONNECT...");
}
```

### Task 5: Update main.cpp

Conditionally include the right driver and assign LED/buzzer pins that don't conflict with HUB75 color data lines (GPIO 25/26/27).

**Files:**
- Modify: `src/main.cpp`

- [ ] **Step 1: Rewrite main.cpp**

```cpp
// src/main.cpp
#include <Arduino.h>
#include "MockDataProvider.h"
#include "DisplayManager.h"
#include "IDisplayDriver.h"
#include "LedController.h"
#include "Buzzer.h"

#ifdef USE_HUB75
  #include "Hub75Driver.h"
  #define GREEN_LED   32
  #define RED_LED     33
  #define BUZZER_PIN  21
#else
  #include "Max7219Driver.h"
  #define CS_PIN      5
  #define MAX_DEVICES 4
  #define GREEN_LED   25
  #define RED_LED     26
  #define BUZZER_PIN  27
#endif

MockDataProvider* dataProvider;
IDisplayDriver*   displayDriver;
DisplayManager*   displayManager;
LedController*    ledController;
Buzzer*           buzzer;

void setup() {
  Serial.begin(115200);
  Serial.println("=== Pickleball Court Prompter (MVP) ===");
  dataProvider = new MockDataProvider();
#ifdef USE_HUB75
  displayDriver = new Hub75Driver();
#else
  displayDriver = new Max7219Driver(CS_PIN, MAX_DEVICES);
#endif
  displayManager = new DisplayManager(*displayDriver, *dataProvider);
  ledController  = new LedController(GREEN_LED, RED_LED);
  buzzer         = new Buzzer(BUZZER_PIN);
  ledController->begin();
  buzzer->begin();
  displayManager->begin();
  buzzer->shortBeep();
  Serial.println("Setup complete. Starting demo...");
}

void loop() {
  DisplayState state = dataProvider->getCurrentState();
  if      (state.status == "PREPARING")   { ledController->greenOn();   ledController->redOff(); }
  else if (state.status == "RUNNING")     { ledController->greenOff();  ledController->redOn(); }
  else if (state.status == "MAINTENANCE") { ledController->greenBlink(); ledController->redOff(); }
  else if (state.status == "OFFLINE")     { ledController->greenOff();  ledController->redBlink(); }
  else                                    { ledController->allOff(); }
  ledController->update();
  buzzer->update();
  displayManager->update();
}
```
